'''Write a python function to find the sum of all odd natural numbers within the range l and r.'''
def sum_odd_numbers(l, r):
    '''
    l: lower bound
    r: upper bound
    '''
    sum = 0
    for i in range(l, r+1):
        if i % 2 == 1:
            sum += i
    return sum
