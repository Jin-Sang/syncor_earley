'''Write a function to find the first adverb ending with ly and its positions in a given string.'''
def find_adverb(string):
    adverb_list = []
    for i in range(len(string)):
        if string[i] == 'y' and string[i+1] == 'l':
            adverb_list.append(i)
    return adverb_list
