def common_elements(list1, list2):
    '''
    list1: list of elements in first list
    list2: list of elements in second list
    returns: True if both lists have at least one common element
    '''
    common = []
    for i in list1:
        if i in list2:
            common.append(i)
    return len(common) > 0
