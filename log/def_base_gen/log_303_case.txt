    if len(arr) == 0:
        return -1
    if k == 0:
        return arr[k]
    if k > len(arr) - 1:
        return findKthElement(arr, k - len(arr) + 1)
    if arr[k] < arr[k - 1]:
        return findKthElement(arr, k - 1)
    return arr[k - 1]
